    	sprite0 sprite1   > transformTo > transformTo stype=sprite1 scoreChange=-3

    	sprite8 sprite0  > killSprite scoreChange=4
    	sprite0 sprite2  > stepBack
    	sprite0 sprite2  > transformTo stype=sprite2 scoreChange=2

    TerminationSet
    	SpriteCounter stype=sprite0 limit=0 win=False
    	Timeout limit=500 win=True
