    	sprite1 > MovingAvatar speed=1.0 img=newset/camel5
    	sprite2 > Immovable color=GREEN color=GRAY
    	sprite3 > Immovable color=BLACK img=newset/cheese
    	sprite4 > Immovable color=GREEN img=oryx/cspell4 orientation=UP color=RED
    	sprite5 > Immovable img=oryx/wall3

    LevelMapping
    	A > sprite0
    	B > sprite1
    	C > sprite2
    	D > sprite3
    	E > sprite4
    	F > sprite5

    InteractionSet
    	sprite1 sprite2   > transformTo > transformTo stype=sprite2 scoreChange=-3
    	sprite4 sprite1   > killSprite > killSprite scoreChange=1
    	sprite2 sprite1  > killSprite scoreChange=-1
    	sprite3 sprite2  > stepBack

    TerminationSet
    	SpriteCounter stype=sprite1 limit=0 win=True
    	SpriteCounter stype=sprite1 limit=0 win=False
    	Timeout limit=1500 win=False
    	Timeout limit=500 win=True
    	Timeout limit=1000 win=True
    	Timeout limit=1000 win=False
